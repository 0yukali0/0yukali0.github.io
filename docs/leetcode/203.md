# solution
新增虛擬頭，確認下個節點與目標直是否相同，持續推進即可。
```
func removeElements(head *ListNode, val int) *ListNode {
    dummyNode := &ListNode{
        Val: -1,
        Next: head,
    }
    node := dummyNode
    for node != nil && node.Next != nil {
        if node.Next.Val == val {
            tmp := node.Next
            node.Next = tmp.Next
            tmp.Next = nil
        } else {
            node = node.Next 
        }
    }
    headNode := dummyNode.Next
    dummyNode.Next = nil
    return headNode
}
```
